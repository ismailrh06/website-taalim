\section{Notion de primitive}

\begin{definition}
Soit $f$ une fonction dÃĐfinie sur un intervalle $I$. Une fonction $F$, dÃĐrivable sur $I$, est une \textbf{primitive de $f$ sur $I$} lorsque
\[

\]
\end{definition}

\begin{propriete}[Existence]
Toute fonction \textbf{continue} sur un intervalle $I$ admet des primitives sur $I$ (thÃĐorÃĻme admis en terminale ; il sera justifiÃĐ en analyse plus poussÃĐe via le calcul intÃĐgral).
\end{propriete}

\begin{propriete}[Structure de l'ensemble des primitives]
Si $F$ est une primitive de $f$ sur $I$, alors les primitives de $f$ sur $I$ sont exactement les fonctions $x\mapsto F(x)+C$, $C\in\mathbb{R}$.
\end{propriete}

\begin{demo}
\emph{Sens direct} : si $G(x)=F(x)+C$, alors $G'(x)=F'(x)=f(x)$ : $G$ est bien une primitive de $f$.

\smallskip
\emph{RÃĐciproque} : soit $G$ une autre primitive de $f$ sur $I$. Posons $\varphi = G - F$. Alors $\varphi'(x) = G'(x)-F'(x) = f(x)-f(x) = 0$ pour tout $x\in I$. \end{demo}

\begin{propriete}[Condition initiale]
Pour tout $(x_0,y_0)$ avec $x_0\in I$, il existe une \textbf{unique} primitive $F$ de $f$ sur $I$ telle que $F(x_0)=y_0$.
\end{propriete}

\begin{demo}
Soit $F_0$ une primitive quelconque de $f$ sur $I$. Toute primitive s'ÃĐcrit $F=F_0+C$. La condition $F(x_0)=y_0$ donne $C=y_0-F_0(x_0)$, valeur \textbf{unique} : la primitive cherchÃĐe est $F=F_0+\bigl(y_0-F_0(x_0)\bigr)$.
\end{demo}

\begin{exemple}[Primitive avec condition initiale]
DÃĐterminons la primitive $F$ de $f(x)=3x^2-4x+1$ sur $\mathbb{R}$ telle que $F(1)=2$.

\smallskip
Les primitives de $f$ sont $F(x)=x^3-2x^2+x+C$. La condition $F(1)=2$ donne $1-2+1+C=2\iff C=2$.

\smallskip
\textbf{Conclusion} : $F(x)=x^3-2x^2+x+2$.
\end{exemple}

\begin{remarque}[La condition ÂŦ sur un intervalle Âŧ est essentielle]
Une primitive se dÃĐfinit toujours \textbf{sur un intervalle}. Sur $\mathbb{R}^*$ (rÃĐunion de deux intervalles, pas un intervalle), la ÂŦ constante Âŧ de l'ÃĐnoncÃĐ peut ÃŠtre \textbf{diffÃĐrente} sur $]-\infty\,;0[$ et sur $]0\,;+\infty[$ : par exemple, les primitives de $\frac1x$ sur $\mathbb{R}^*$ sont $\ln|x|+C_1$ sur $]-\infty\,;0[$ et $\ln|x|+C_2$ sur $]0\,;+\infty[$, avec $C_1$ et $C_2$ \textbf{indÃĐpendantes}.
