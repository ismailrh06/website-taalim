\section{DivisibilitÃĐ dans $\mathbb{Z}$}

\begin{definition}[DivisibilitÃĐ]
Soient $a,b\in\mathbb{Z}$. On dit que $b$ \textbf{divise} $a$ (ou que $a$ est un \textbf{multiple} de $b$, ou que $b$ est un \textbf{diviseur} de $a$), notÃĐ $b\mid a$, lorsqu'il existe $k\in\mathbb{Z}$ tel que $a=kb$.
\end{definition}

\begin{propriete}[PropriÃĐtÃĐs de la divisibilitÃĐ]
Pour $a,b,c\in\mathbb{Z}$ :
\begin{itemize}[leftmargin=6mm, itemsep=0.5mm]
  \item \textbf{RÃĐflexivitÃĐ} : $a\mid a$. \quad Tout entier divise $0$ ; $1$ divise tout entier.
  \item \textbf{TransitivitÃĐ} : si $a\mid b$ et $b\mid c$, alors $a\mid c$.
  \item Si $a\mid b$ et $a\mid c$, alors $a\mid(ub+vc)$ pour tous $u,v\in\mathbb{Z}$ (\textbf{combinaison linÃĐaire}).
  \item Si $a\mid b$ et $b\mid a$ (avec $a,b\ne0$), alors $a=\pm b$.
  \item Si $a\mid b$ et $a,b\ne0$, alors $|a|\leq|b|$.
\end{itemize}
\end{propriete}

\begin{demo}
\emph{TransitivitÃĐ} : $a\mid b\Rightarrow b=ka$ ; $b\mid c\Rightarrow c=k'b=k'ka$ : donc $a\mid c$ avec le coefficient $kk'$.

\smallskip
\emph{Combinaison linÃĐaire} : $a\mid b\Rightarrow b=ka$ ; $a\mid c\Rightarrow c=k'a$. Alors $ub+vc=uka+vk'a=(uk+vk')a$ : $a$ divise bien $ub+vc$.

\smallskip
\emph{$a\mid b$ et $b\mid a\Rightarrow a=\pm b$} : $b=ka$ et $a=k'b=k'ka$, donc (si $a\ne0$) $kk'=1$, soit $k=k'=1$ ou $k=k'=-1$ (entiers de produit $1$) : $b=a$ ou $b=-a$.
\end{demo}
